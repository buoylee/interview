```javascript
devServer: {
  host: '0.0.0.0',
  port: 8081,
  allowedHosts: 'all',
  https: (() => {
    // ... 原有 HTTPS 配置
  })(),
  client: {

  // 🆕 新增：代理配置
  proxy: {
    '/api/getcookieuserid': {
      target: 'https://beta.api.cocorobo.cn',
      changeOrigin: true,
      secure: false,
      cookieDomainRewrite: {
        '*': ''
      },
      onProxyReq: function(proxyReq, req, res) {
        console.log('🔄 代理请求:', req.url, '→', 'https://beta.api.cocorobo.cn' + req.url);
      },
      onProxyRes: function(proxyRes, req, res) {
        const setCookieHeaders = proxyRes.headers['set-cookie'];
        if (setCookieHeaders) {
          proxyRes.headers['set-cookie'] = setCookieHeaders.map(cookie => {
            return cookie
              .replace(/; Secure/gi, '')
              .replace(/; SameSite=\w+/gi, '');
          });
        }
        console.log('✅ 代理响应:', req.url, '状态码:', proxyRes.statusCode);
      }
    }
  }
}
```

**影响范围**：
- 开发环境本地调试
- 不影响生产环境

### 3. src/common/mixin.js

**修改位置**：第 1033-1058 行（getUrlParams 方法）

**修改内容**：
- 修复 hash 路由参数读取问题
- 支持从 `#/?id=xxx&type=xxx` 格式中提取参数

**修改前**：
```javascript
getUrlParams(url) {
  const params = {};
  const urlObj = new URL(url);
  const queryString = urlObj.search.slice(1);
  const queryPairs = queryString.split('&');

  queryPairs.forEach(pair => {
    const [key, value] = pair.split('=');
    if (key) {
      params[decodeURIComponent(key)] = decodeURIComponent(value || '');
    }
  });

  return params;
}
```

**修改后**：
```javascript
getUrlParams(url) {
  const params = {};
  const urlObj = new URL(url);

  // 尝试从标准查询字符串获取参数（?id=...&type=...）
  let queryString = urlObj.search.slice(1);

  // 🆕 如果没有查询字符串，尝试从 hash 中获取（#/?id=...&type=...）
  if (!queryString && urlObj.hash) {
    const hashParts = urlObj.hash.split('?');
    if (hashParts.length > 1) {
      queryString = hashParts[1];
    }
  }

  const queryPairs = queryString.split('&');

  queryPairs.forEach(pair => {
    const [key, value] = pair.split('=');
    if (key) {
      params[decodeURIComponent(key)] = decodeURIComponent(value || '');
    }
  });

  return params;
}
```

**影响范围**：
- 全局方法，影响所有使用 `getUrlParams` 的地方
- 向后兼容：原有 `?id=xxx` 格式仍然有效
- 新增支持：现在也支持 `#/?id=xxx` 格式

## 🔧 系统配置修改

### 4. /etc/hosts

**修改内容**：
```
127.0.0.1  local.cocorobo.cn
```

**作用**：
- 将域名 `local.cocorobo.cn` 解析到本地 IP
- 使浏览器能够访问本地开发服务器，并匹配 `.cocorobo.cn` Cookie 域

**恢复方法**：
```bash
sudo nano /etc/hosts
# 删除或注释掉 local.cocorobo.cn 那一行
```
